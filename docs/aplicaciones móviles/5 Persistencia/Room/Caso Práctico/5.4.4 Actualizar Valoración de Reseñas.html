import android.graphics.Bitmap;
import android.graphics.BitmapFactory;
import android.os.Bundle;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;

import androidx.annotation.NonNull;
import androidx.annotation.Nullable;
import androidx.fragment.app.Fragment;

import com.agora.criticascine.databinding.FragmentDetallePeliculaBinding;
import com.agora.criticascine.Model.Pelicula;

public class FragmentDetallePelicula extends Fragment {

    private FragmentDetallePeliculaBinding binding;

    @Nullable
    @Override
    public View onCreateView(@NonNull LayoutInflater inflater, @Nullable ViewGroup container, @Nullable Bundle savedInstanceState) {
        // Vincula el dise√±o con ViewBinding.
        binding = FragmentDetallePeliculaBinding.inflate(inflater, container, false);
        return binding.getRoot();
    }

    @Override
    public void onViewCreated(@NonNull View view, @Nullable Bundle savedInstanceState) {
        super.onViewCreated(view, savedInstanceState);

        // Obtiene los argumentos pasados al fragmento.
        if (getArguments() != null) {
            
            Pelicula pelicula = (Pelicula) getArguments().getSerializable("pelicula");

            // Configura los datos de la rese√±a en las vistas.
            binding.textoNombreDetalle.setText(pelicula.getNombre());
            binding.textoDescripcionDetalle.setText(pelicula.getDescripcion());
            binding.barraValoracionDetalle.setRating(pelicula.getValoracion());

            // Convierte el array de bytes a Bitmap y lo muestra.
            if (pelicula.getImagen() != null) {
                Bitmap bitmap = BitmapFactory.decodeByteArray(pelicula.getImagen(), 0, pelicula.getImagen().length);
                binding.imagenDetalle.setImageBitmap(bitmap);
            }
        }
    }

    @Override
    public void onDestroyView() {
        super.onDestroyView();
        binding = null; // Libera el binding para evitar fugas de memoria.
    }
}
