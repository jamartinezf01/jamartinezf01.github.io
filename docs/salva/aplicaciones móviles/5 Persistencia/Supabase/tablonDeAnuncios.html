5.5.3 Tablón de Anuncios

Vamos a añadir una nueva funcionalidad a la aplicación Tablón de Anuncios creada en el punto anteriro (Firestore)

Para añadir imágenes almacenadas en la nube a los anuncios de nuestro tablón vamos anecesitar las siguientes modificaciones...

1. Actualizar el Modelo​

Deberemos añadir a nuestra clase Anuncio los atributos y métodos necesarios para almacenar la URL de la imagen del anuncio.

model/Anuncio.java
package com.agora.tablonfirestore.model;

public class Anuncio {
    private String id;
    private String contenido;
    private long fecha;
    private String uidAutor;
    private String imagen;

    public Anuncio() {} // Constructor vacío para Firebase

    public Anuncio(String id, String contenido, long fecha, String uidAutor) {
        this.id = id;
        this.contenido = contenido;
        this.fecha = fecha;
        this.uidAutor = uidAutor;
    }

    public String getId() { return id; }
    public String getContenido() { return contenido; }
    public long getFecha() { return fecha; }
    public String getUidAutor() { return uidAutor; }

    public void setImagen(String imagen) {
        this.imagen = imagen;
    }

    public String getImagen() {
        return imagen;
    }
}

2. Crear una clase para el servicio Supabase​

En nuestro modelo, crearemos una clase para manejar las peticiones a la API de supabase. Podemos llamar a esta clase SupabaseStorageService ya que se encargará de las peticiones al servicio de storage.

model/SupabaseStorageService.java
package com.agora.tablonfirestore.model;

import android.content.ContentResolver;
import android.content.Context;
import android.net.Uri;
import android.util.Log;

import okhttp3.*;

import java.io.ByteArrayOutputStream;
import java.io.IOException;
import java.io.InputStream;

/*
 * Esta clase se encarga de gestionar la subida de imágenes
 * a un bucket de almacenamiento en Supabase.
 */
public class SupabaseStorageService {

    // URL base de tu proyecto en Supabase. Aquí se indican el protocolo (https),
    // la subdominio y el dominio con la ruta del proyecto.
    private static final String SUPABASE_URL = "https://nilzuhkcvqjiuvylgmqd.supabase.co";

    // Clave de acceso a Supabase. Es necesaria para autorizar las peticiones.
    // Esta clave suele encontrarse en las configuraciones del proyecto de Supabase.
    private static final String SUPABASE_KEY = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6Im5pbHp1aGtjdnFqaXV2eWxnbXFkIiwicm9sZSI6ImFub24iLCJpYXQiOjE3Mzg4NzAxNzQsImV4cCI6MjA1NDQ0NjE3NH0.p0bevRZTdbAJgeetpUUfj_eH_R47IFqSu-xDqJcyZls";
    
    // Nombre del bucket donde se almacenarán las imágenes
    // (tal como se haya configurado en Supabase Storage).
    private static final String BUCKET = "imagenes";

    // Cliente OkHttp para realizar peticiones HTTP.
    private final OkHttpClient client = new OkHttpClient();

    // Contexto de la aplicación. Se necesita para acceder al ContentResolver
    // y poder abrir el InputStream de la imagen.
    private final Context context;

    /*
     * Interfaz para manejar los resultados de la subida de la imagen.
     * - onSuccess: si la subida fue exitosa, se pasa la URL pública de la imagen.
     * - onFailure: si ocurre algún error, se notifica con el mensaje correspondiente.
     */
    public interface UploadCallback {
        void onSuccess(String url);
        void onFailure(String error);
    }

    /*
     * Constructor que recibe el contexto para poder trabajar con
     * recursos de Android (como el ContentResolver).
     */
    public SupabaseStorageService(Context context) {
        this.context = context;
    }

    /*
     * Método principal para subir una imagen.
     * uri: la ubicación de la imagen en el dispositivo.
     * userEmail: se utiliza como parte de la ruta de guardado en Supabase (subcarpeta).
     * anuncioId: nombre o identificador del archivo (por ejemplo, "fotoDeAnuncio.jpg").
     * callback: manejo de éxito o error.
     */
    public void subirImagen(Uri uri, String userEmail, String anuncioId, UploadCallback callback) {
        try {
            // Se obtiene el ContentResolver para acceder a la imagen a partir de la uri.
            ContentResolver contentResolver = context.getContentResolver();

            // Se abre un InputStream para leer los bytes de la imagen.
            InputStream inputStream = contentResolver.openInputStream(uri);
            if (inputStream == null) {
                // Si por algún motivo no se pudo abrir el InputStream, se notifica error y se sale.
                callback.onFailure("No se pudo abrir el InputStream");
                return;
            }

            // Se convierten todos los datos del InputStream (imagen) a un array de bytes.
            byte[] imageData = convertirInputStreamABytes(inputStream);

            // Se define la ruta del archivo dentro del bucket en Supabase.
            // Por ejemplo, se guardará en "usuario@example.com/1234.jpg".
            String filePath = userEmail + "/" + anuncioId + ".jpg";

            // Se construye el cuerpo de la petición (multipart/form-data)
            // donde se adjunta la imagen bajo la clave "file".
            RequestBody requestBody = new MultipartBody.Builder()
                    .setType(MultipartBody.FORM)
                    .addFormDataPart(
                            "file",            // Nombre del campo en el formulario
                            filePath,          // Nombre del archivo
                            // Crea un RequestBody específico para la imagen con tipo "image/jpeg".
                            RequestBody.create(MediaType.parse("image/jpeg"), imageData)
                    )
                    .build();

            // Construcción de la URL completa hacia donde se hará el PUT.
            // Formato: https://<URL>/storage/v1/object/<bucket>/<ruta_archivo>
            String url = SUPABASE_URL + "/storage/v1/object/" + BUCKET + "/" + filePath;

            // Creación de la petición HTTP tipo PUT, con las cabeceras necesarias:
            // - Authorization: se pasa la 'Bearer' token para la API de Supabase.
            // - Content-Type: "image/jpeg".
            Request request = new Request.Builder()
                    .url(url)
                    .header("Authorization", "Bearer " + SUPABASE_KEY)
                    .header("Content-Type", "image/jpeg")
                    .put(requestBody)
                    .build();

            // Se lanza la operación en un hilo aparte para no bloquear el hilo principal de la app.
            new Thread(() -> {
                try {
                    // Se ejecuta la llamada al servidor.
                    Response response = client.newCall(request).execute();
                    
                    // Se comprueba si la respuesta fue exitosa (código 2xx).
                    if (response.isSuccessful()) {
                        // Si tuvo éxito, se construye la URL pública de la imagen.
                        // La carpeta "public" se asume habilitada en la política del bucket.
                        String publicUrl = SUPABASE_URL + "/storage/v1/object/public/" + BUCKET + "/" + filePath;
                        callback.onSuccess(publicUrl);
                    } else {
                        // Si ocurre algún error, se lee el cuerpo de la respuesta para informarlo.
                        String responseBody = response.body() != null ? response.body().string() : "Respuesta vacía";
                        Log.e("Supabase", "Error al subir imagen: " + response.code() + " - " + responseBody);
                        callback.onFailure("Error " + response.code() + ": " + responseBody);
                    }
                } catch (Exception e) {
                    // Cualquier excepción se registra y se pasa al callback.
                    Log.e("Supabase", "Excepción al subir imagen: " + e.getMessage(), e);
                    callback.onFailure("Error de conexión: " + e.getMessage());
                }
            }).start();

        } catch (Exception e) {
            // Si se produce un error al procesar los datos (por ejemplo, leer la imagen),
            // se notifica al callback.
            callback.onFailure("Error al procesar la imagen: " + e.getMessage());
        }
    }

    /*
     * Convierte un InputStream en un array de bytes.
     * Esto se utiliza para poder adjuntar la imagen en el cuerpo de la petición.
     */
    private byte[] convertirInputStreamABytes(InputStream inputStream) throws IOException {
        ByteArrayOutputStream buffer = new ByteArrayOutputStream();
        // Se crea un array de lectura de 1KB. Se leerá en bloques.
        byte[] data = new byte[1024];
        int bytesRead;
        // Se leen los bytes del InputStream y se guardan en el buffer hasta agotar el stream.
        while ((bytesRead = inputStream.read(data, 0, data.length)) != -1) {
            buffer.write(data, 0, bytesRead);
        }
        // Se devuelven todos los bytes leídos en un array.
        return buffer.toByteArray();
    }

}

/*
 * Actualiza la URL de la imagen en un documento "Anuncio" específico dentro de Firestore.
 */
    public void actualizarImagenAnuncio(String anuncioId, String imagenUrl, AuthCallback callback) {
        // anunciosRef hace referencia a una colección en Firestore que contiene documentos de "Anuncios".
        // Cada documento de "Anuncio" posee un ID, en este caso "anuncioId".
        anunciosRef.document(anuncioId)
                // Se actualiza únicamente el campo "imagen" del documento con la nueva URL (imagenUrl).
                .update("imagen", imagenUrl)
                // addOnSuccessListener: se ejecuta si la operación de actualización fue exitosa.
                .addOnSuccessListener(aVoid -> {
                    // Se registra en el Logcat de Android un mensaje de éxito.
                    Log.d("Firestore", "Imagen actualizada en el anuncio");
                    // Se llama al método onSuccess() del callback para notificar que todo salió bien.
                    callback.onSuccess();
                })
                // addOnFailureListener: se ejecuta si ocurre algún error durante la operación de actualización.
                .addOnFailureListener(e -> {
                    // Se registra el error en el Logcat.
                    Log.e("Firestore", "Error al actualizar la imagen", e);
                    // Se llama al método onFailure() del callback y se envía el mensaje de error.
                    callback.onFailure(e.getMessage());
                });
    }

        // ...

        // Si hay una imagen en el anuncio, se carga en el ImageView usando Glide

Modificar el Fragmento que implementa el ReciclerView.​

Para finalizar deberemos actualizar el Fragment TablonFragment para poder abrir la galería y seleccionar una imagen al pulsar el botón correspondiente del anuncio.

/ui/TablonFragment.java

public class TablonFragment extends Fragment {

    //...

    @Override
    public void onViewCreated(@NonNull View view, @Nullable Bundle savedInstanceState) {
        
        // ...

        // Variable para almacenar temporalmente el anuncio seleccionado
        private Anuncio anuncioSeleccionado;

        /*
        * Launcher para la selección de imágenes desde la galería.
        * Al obtener el resultado (URI de la imagen), se llama a subirImagenAStorage().
        */
        private ActivityResultLauncher<Intent> seleccionarImagenLauncher =
                registerForActivityResult(new ActivityResultContracts.StartActivityForResult(),
                        result -> {
                            // Verificamos que la operación se haya completado con éxito
                            if (result.getResultCode() == Activity.RESULT_OK && result.getData() != null) {
                                Uri imagenUri = result.getData().getData();
                                // Si hay una imagen y tenemos un anuncio seleccionado, subimos la imagen a Supabase
                                if (imagenUri != null && anuncioSeleccionado != null) {
                                    subirImagenAStorage(imagenUri, anuncioSeleccionado.getId(), anuncioSeleccionado.getUidAutor());
                                }
                            }
                        });

        /*
        * Método que se invoca al pulsar el botón de "agregar imagen" en el adaptador.
        * Recibe el ID del anuncio y el email del autor.
        * Lanza un intent para abrir la galería y obtener la imagen seleccionada.
        */
        public void seleccionarImagen(String anuncioId, String userEmail) {
            // Creamos un objeto Anuncio "dummy" con el ID y usuario para guardarlo temporalmente
            this.anuncioSeleccionado = new Anuncio(anuncioId, "", 0, userEmail);

            // Creamos un Intent para abrir la galería (ACTION_PICK) y filtrar solo imágenes
            Intent intent = new Intent(Intent.ACTION_PICK);
            intent.setType("image/*");
            // Iniciamos el ActivityResultLauncher
            seleccionarImagenLauncher.launch(intent);
        }

        /*
        * Se encarga de subir la imagen a Supabase Storage:
        *   - uri: ubicación de la imagen seleccionada
        *   - anuncioId y userEmail: para construir la ruta/nombre de la imagen en el Storage
        */
        private void subirImagenAStorage(Uri uri, String anuncioId, String userEmail) {
            // Instanciamos el servicio de Supabase, pasando el contexto
            SupabaseStorageService storageService = new SupabaseStorageService(requireContext());

            // Llamamos al método subirImagen con la URI y los datos del anuncio
            storageService.subirImagen(uri, userEmail, anuncioId, new SupabaseStorageService.UploadCallback() {
                @Override
                public void onSuccess(String url) {
                    // Cuando se completa la subida, actualizamos el Anuncio en Firestore con la URL
                    if (tablonVM != null) {
                        tablonVM.actualizarImagenAnuncio(anuncioId, url);
                    }
                }

                @Override
                public void onFailure(String error) {
                    // Si falla, registramos el error en el Logcat
                    Log.e("Supabase", "Error al subir imagen: " + error);
                }
            });
        }

    }
}
